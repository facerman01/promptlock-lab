# windows !?
    # 1. Disable the PowerShell Progress Bar
    $ProgressPreference = 'SilentlyContinue'

    # 2. Define your variables
    $url = "https://github.com/ollama/ollama/releases/latest/download/ollama-windows-amd64.zip"
    $zipPath = "$env:TEMP\ollama.zip"

    # 3. Use the raw .NET framework for maximum speed
    Write-Host "Starting high-speed download..."
    $webClient = New-Object System.Net.WebClient
    $webClient.DownloadFile($url, $zipPath)
    Write-Host "Download complete!"

    # 4. Set the destination folder
    $installDir = "C:\Ollama"

    # 5. Extract the downloaded zip file
    Write-Host "Extracting files to $installDir..."
    Expand-Archive -Path "$env:TEMP\ollama.zip" -DestinationPath $installDir -Force

    # 6. Add to the current session's PATH so Windows knows what 'ollama' is
    $env:Path += ";$installDir"
    
    # 6.1 Add to path permanently
    $sysPath = [Environment]::GetEnvironmentVariable("Path", "Machine")
    [Environment]::SetEnvironmentVariable("Path", "$sysPath;C:\Ollama", "Machine")

    # 6.2 Start the Engine
    Start-Process -FilePath "C:\Ollama\ollama.exe" -ArgumentList "serve" -WindowStyle Hidden

    # 7. Verify it works!
    ollama --version
